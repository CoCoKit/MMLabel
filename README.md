###Label 关键字高亮
>关键字高亮显示，可设置关键字颜色、大小

>//关键词

> @property(nonatomic,strong) NSString *keyWord;
> 
> //关键词颜色
> 
> @property(nonatomic,strong) UIColor *keyWordColor;
> 
> //关键词字体
> 
>@property(nonatomic,strong) UIFont *keyWordFont;
